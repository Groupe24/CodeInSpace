\frame{\frametitle{Cette année : dévelopement d'un jeu et d'une IA}
	\begin{block}{Le projet en quelques mots}
		par groupes (libres !) de 3 étudiants, implémenter
		\begin{itemize}
			\item un programme permettant de jouer à un jeu raisonnablement complexe
			\item une intelligence artificielle (IA) contre laquelle jouer à ce jeu
		\end{itemize}
		\vspace{.2cm}
		
		le jeu était imposé $\Rightarrow$ découverte de l'énoncé en groupe début février
	\end{block}
	\vspace{.2cm}
	
	\begin{block}{Organisation du quadrimestre}
		\begin{itemize}
			\item le laboratoire est l'aboutissement d'INFOB131 et des mini-projets
			\item uniquement des séances en groupe encadrées par les assistants
			\item pas assez $\Rightarrow$ cours de 3 ECTS = 90 heures de travail/étudiant
			\item les étapes du laboratoire étaient soigneusement balisées
%			\item<3-> il était \textbf{impossible} de faire le projet tout seul : prévu pour 3 personnes
		\end{itemize}
	\end{block}

\begin{frame}[fragile]
	\frametitle{Quelques détails sur l'intelligence artificielle}
	\begin{block}{}
		intelligence artificielle = joueur dont les coups sont choisis par l'ordinateur automatiquement en fonction de l'état actuel du jeu et d'une stratégie
	\end{block}
	
	\begin{block}{}
		\begin{lstlisting}
			def get_IA_orders(game, player_id):
			    """Returns orders chosen by IA player.
			    
			    Parameters
			    ----------
			    game: game data structure (dict)
			    player_id: id of player, 1 or 2 (int)
			    
			    Returns
			    -------
			    orders: orders chosen by IA player (str)
			    
			    """
		\end{lstlisting}
	\end{block}
\end{frame}

	\begin{block}{Déroulement du tournoi}
		\begin{itemize}
			\item tournoi en phases: premières phases BO3 et dernières phases BO5
			\begin{itemize}
				\item best of 3 = premier joueur à gagner 2 parties (3 parties max.)
				\item best of 5 = premier joueur à gagner 3 parties (5 parties max.)
			\end{itemize}
			\item à chaque partie, le premier joueur et la taille du plateau sont aléatoires
			\item IA bloquée (60 sec. max.) ou crashée (exception) $\Rightarrow$ partie perdue
			\item grande et petite finale pour déterminer les trois premières places
		\end{itemize}
	\end{block}	
\end{frame}
